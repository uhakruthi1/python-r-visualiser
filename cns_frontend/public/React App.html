# Loading the ggplot2 package
library(ggplot2)
# Creating a data frame
data <- data.frame(
  category = c("A", "B", "C", "D"),
  value = c(23, 17, 35, 29)
)
# Creating a bar plot
ggplot(data, aes(x = category, y = value)) +
  geom_bar(stat = "identity", fill = "steelblue") +
  labs(title = "Sample Bar Plot", x = "Category", y = "Value")
